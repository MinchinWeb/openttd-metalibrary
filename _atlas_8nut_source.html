269 function _MinchinWeb_Atlas_::RunModel() {
270 // Takes the provided sources and destinations and runs the selected traffic model, populating the 'pairs' heap
271  this._pairs = this._heap_class();
272  for (local i = 0; i < this._sources.len(); i++) {
273  _MinchinWeb_Log_.Note(" i = " + i, 7);
274  for (local j = 0; j < this._attractions.len(); j++) {
275  _MinchinWeb_Log_.Note(" j = " + j + " " + (_MinchinWeb_Array_.ContainedIn1D([this._sources[i][0], this._attractions[j][0]], this._ignorepairs) == false) + "; " + _MinchinWeb_Atlas_.ApplyTrafficModel(this._sources[i][0], 1, this._attractions[j][0], 0, this._maxdistancemodel) + " < " + this._maxdistance + " = " + ((_MinchinWeb_Atlas_.ApplyTrafficModel(this._sources[i][0], 1, this._attractions[j][0], 0, this._maxdistancemodel) < this._maxdistance) || (this._maxdistance < 0)) + " ;; " + _MinchinWeb_Array_.ToStringTiles1D([this._sources[i][0]]) + " - " + this._sources[i][1] + " " + _MinchinWeb_Array_.ToStringTiles1D([this._attractions[j][0]]) + " - " + this._attractions[j][1] + " -- " + this._model + " :: " + AIMap.DistanceManhattan(this._sources[i][0], this._attractions[j][0]) + " / (" + this._sources[i][1] + " + " + this._attractions[j][1] + ") = " + (AIMap.DistanceManhattan(this._sources[i][0], this._attractions[j][0]) / (this._sources[i][1].tofloat() + this._attractions[j][1].tofloat())) + " : " + _MinchinWeb_Atlas_.ApplyTrafficModel(this._sources[i][0], this._sources[i][1], this._attractions[j][0], this._attractions[j][1], this._model), 7);
276  if ((_MinchinWeb_Array_.ContainedIn1D([this._sources[i][0], this._attractions[j][0]], this._ignorepairs) == false)
277  && ((_MinchinWeb_Atlas_.ApplyTrafficModel(this._sources[i][0], 1, this._attractions[j][0], 0, this._maxdistancemodel) < this._maxdistance)
278  || (this._maxdistance < 0))) {
279  this._pairs.Insert([this._sources[i][0], this._attractions[j][0]],
280  _MinchinWeb_Atlas_.ApplyTrafficModel(this._sources[i][0], this._sources[i][1], this._attractions[j][0], this._attractions[j][1], this._model));
281  }
282  }
283  }

340 function _MinchinWeb_Atlas_::ApplyTrafficModel(StartTile, StartPriority, EndTile, EndPriority, Model) {
341 // Given the start and end points, applies the traffic model and returns the
342 // weighting
343 // Smaller weightings are considered better
344 
345  switch (Model) {
346  case ModelType.ONE_D :
347  return AIMap.DistanceMax(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
348  break;
349  case ModelType.DISTANCE_MANHATTAN :
350  return AIMap.DistanceManhattan(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
351  break;
352  case ModelType.DISTANCE_SHIP :
353  return _MinchinWeb_Marine_.DistanceShip(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
354  break;
355  case ModelType.DISTANCE_AIR :
356  return AIMap.DistanceSquare(StartTile, EndTile) / (StartPriority.tofloat() + EndPriority.tofloat());
357  break;
358  case ModelType.DISTANCE_NONE :
359  return (1.0 / (StartPriority.tofloat() + EndPriority.tofloat()));
360  break;
361  case ModelType.ONE_OVER_T_SQUARED :
362  return ((AIMap.DistanceManhattan(StartTile, EndTile) * AIMap.DistanceManhattan(StartTile, EndTile)) / (StartPriority.tofloat() + EndPriority.tofloat()));
363  break;
364  default:
365  return "ERROR: Bad ModelType. Was supplied: " + Model;
366  break;
367  }
368 }
