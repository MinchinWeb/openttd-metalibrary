397 function _MinchinWeb_Lakes_::AddPoint(myTileID) {
398  // _MinchinWeb_Log_.Note("Tile Area " + this._map.GetValue(myTileID) + " : " + this._map.Count() + " / " + this._areas.len(), 6);
399 
400  switch (this._map.GetValue(myTileID)) {
401  case -2:
402  // tile is unset (i.e. in no group)
403  if (AITile.IsWaterTile(myTileID) == true) {
404  // add to _map if a water tile
405  local myArea = this._areas.len();
406  this._areas.append(myTileID);
407  this._open_neighbours.append([]);
408  this._connections.append([]);
409  this._map.SetValue(myTileID, myArea);
410  this._group_tiles.append(AIList());
411  this._group_tiles[myArea].AddItem(myTileID, myTileID);
412 
413  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
414  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
415 
416  foreach (offset in offsets) {
417  local next_tile = myTileID + offset;
418  if (AIMarine.AreWaterTilesConnected(myTileID, next_tile)) {
419  if (this._map.GetValue(next_tile) == -2) {
420  this._open_neighbours[myArea].append([myTileID, next_tile]);
421  } else if (this._map.GetValue(next_tile) == -1) {
422  // the tile has ceased to be land (somehow...)
423  this._map.SetValue(next_tile, -2);
424  this._open_neighbours[myArea].append([myTileID, next_tile]);
425  } else {
426  // register connection right now
427  local ConnectedArea = this._map.GetValue(next_tile);
428 
429  this._connections[myArea].append(ConnectedArea);
430  this._connections[ConnectedArea].append(myArea);
431 
432  local AllConnectedAreas = _AllGroups([ConnectedArea]);
433 
434  // remove open neighbour from AllConnectedAreas to next_tile
435  foreach (ThisArea in AllConnectedAreas) {
436  for (local i=0; i < this._open_neighbours[ThisArea].len(); i++) {
437  if (this._open_neighbours[ThisArea][i][1] == myTileID) {
438  // we're looking for the reverse tile pair to the one we just tried to add
439  this._open_neighbours[ThisArea] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[ThisArea], i);
440  i--;
441  }
442  }
443  }
444  }
445  }
446  }
447 
448  _MinchinWeb_Log_.Note(myArea + " : " + _MinchinWeb_Array_.ToStringTiles2D(this._open_neighbours[myArea], false), 7);
449  _MinchinWeb_Log_.Sign(myTileID, "L" + myArea, 8);
450  return myArea;
451  } else {
452  // land tile
453  this._map.SetValue(myTileID, -1);
454  // not added to _group_tiles
455  return -1;
456  }
457  case -1:
458  // land tile
459  return -1;
460  default:
461  // already in _map
462  return this._map.GetValue(myTileID);
463  }
464 }

466 function _MinchinWeb_Lakes_::_AllGroups(StartGroupArray) {
467  // this function starts with an array of starting groups
468  // starts at the beginning and for the first group, appends all connected
469  // groups to the end of the array
470  // does the same for the second and so on until the original end of the
471  // array
472  // next it compacts the resulting array by removing duplicates
473  // then it picks up where it left off in the (now larger) array and starts
474  // adding connections again
475  // this cycle keeps going until no more connections are added that aren't
476  // duplicates
477 
478  local loops = 0;
479  local StartIndex = 0;
480  local ReturnGroup = StartGroupArray;
481  local NextStartIndex = 0;
482  local MoreAdded = true;
483 
484  do {
485  //_MinchinWeb_Log_.Note("In AllGroups(), loop " + loops + ". start: " + StartIndex + " // " + _MinchinWeb_Array_.ToString1D(ReturnGroup, false), 6);
486  MoreAdded = false;
487  NextStartIndex = ReturnGroup.len();
488  for (local i = StartIndex; i < NextStartIndex; i++) {
489  if ((ReturnGroup[i] >= 0) && (this._connections[ReturnGroup[i]].len() > 0)) {
490  ReturnGroup = _MinchinWeb_Array_.Append(ReturnGroup, this._connections[ReturnGroup[i]]);
491  ReturnGroup = _MinchinWeb_Array_.RemoveDuplicates(ReturnGroup);
492  MoreAdded = true;
493  }
494  }
495  StartIndex = NextStartIndex;
496  loops++;
497  } while (MoreAdded == true)
498 
499  return ReturnGroup;
500 }

502 function _MinchinWeb_Lakes_::_AddNeighbour(NextTile) {
503  // Start by finding out what area we're expanding
504  local ReturnTiles = array(0);
505  local OnwardTiles = array(0);
506  for (local i = 0; i < this._open_neighbours.len(); i++) {
507  for (local j = 0; j < this._open_neighbours[i].len(); j++) {
508  if (this._open_neighbours[i][j][0] == NextTile) {
509  OnwardTiles.append(this._open_neighbours[i][j][1]);
510  this._open_neighbours[i] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[i], j);
511  j--;
512  }
513  }
514  }
515  // remove duplicates
516  OnwardTiles = _MinchinWeb_Array_.RemoveDuplicates(OnwardTiles);
517  // remove onward tiles that are already in a tile group
518  local ConnectedGroups = _AllGroups([this._map[NextTile]]);
519  for (local i = 0; i < OnwardTiles.len(); i++) {
520  if (this._map[OnwardTiles[i]] != -2) {
521  // But only if we've already registered the connection
522  if (_MinchinWeb_Array_.ContainedIn1D(ConnectedGroups, this._map[OnwardTiles[i]])) {
523  OnwardTiles = _MinchinWeb_Array_.RemoveValueAt(OnwardTiles, i);
524  // Add remove from open neighbours
525  i--;
526  }
527  }
528 
529  }
530  _MinchinWeb_Log_.Note("NextTile: " + _MinchinWeb_Array_.ToStringTiles1D([NextTile]) + " // Onward Tiles: " + _MinchinWeb_Array_.ToStringTiles1D(OnwardTiles), 8);
531 
532  if (OnwardTiles.len() == 0) {
533  // if something broke, spit out useful debug information
534  _MinchinWeb_Log_.Note(" MinchinWeb.Lakes._AddNeighbour(): No source for " + _MinchinWeb_Array_.ToStringTiles1D([NextTile]), 8);
535  /*_MinchinWeb_Log_.Note(" this._open_neighbours");
536  for (local i = 0; i < this._open_neighbours.len(); i++) {
537  _MinchinWeb_Log_.Note(" [" + i + "] " + _MinchinWeb_Array_.ToString2D(this._open_neighbours[i]), 0);
538  }*/

541  } else {

543  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
544  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
545 
546  // Check that each neighbour is still attached by water to our from tile,
547  // and then add it to the FromGroup
548  // Then add possible neighbours to the open_neighbours list
549  foreach (OnwardTile in OnwardTiles) {
550  if (AIMarine.AreWaterTilesConnected(NextTile, OnwardTile)) {
551  this._map.SetValue(OnwardTile, FromGroup);
552  this._group_tiles[FromGroup].AddItem(OnwardTile, OnwardTile);
553  ReturnTiles.append(OnwardTile);
554  _MinchinWeb_Log_.Sign(OnwardTile, "L" + FromGroup, 8);
555  foreach (offset in offsets) {
556  local next_tile = OnwardTile + offset;
557  if (AIMarine.AreWaterTilesConnected(OnwardTile, next_tile) && (this._map.GetValue(next_tile) != this._map.GetValue(OnwardTile))) {
558  this._open_neighbours[FromGroup].append([OnwardTile, next_tile]);
559  }
560  }
561  }
562 
563  // If more than one groups list this tile as an open neighbour,
564  // register the two groups are now linked
565  for (local i = 0; i < this._open_neighbours.len(); i++) {
566  for (local j = 0; j < this._open_neighbours[i].len(); j++) {
567  if (this._open_neighbours[i][j][1] == OnwardTile) {
568  local ActiveFromGroup = this._map.GetValue(this._open_neighbours[i][j][0]);
569  this._connections[FromGroup].append(ActiveFromGroup);
570  this._connections[ActiveFromGroup].append(FromGroup);
571  this._open_neighbours[i] = _MinchinWeb_Array_.RemoveValueAt(this._open_neighbours[i], j);
572  j--;
573
