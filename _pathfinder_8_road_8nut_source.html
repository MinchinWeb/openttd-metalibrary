337  local slope = AITile.GetSlope(end_a);

354 function _MinchinWeb_RoadPathfinder_::_Cost(self, path, new_tile, new_direction)
355 {
356  /* path == null means this is the first node of a path, so the cost is 0. */
357  if (path == null) return 0;
358 
359  local prev_tile = path.GetTile();
360 
361  /* If the new tile is (already) a bridge / tunnel tile, check whether we
362  * came from the other end of the bridge / tunnel or if we just entered the
363  * bridge / tunnel. */
364  if (AIBridge.IsBridgeTile(new_tile)) {
365  if (AIBridge.GetOtherBridgeEnd(new_tile) != prev_tile) {
366  return path.GetCost() + self._cost_tile;
367  } else {
368  return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * self._cost_tile + self._GetBridgeNumSlopes(new_tile, prev_tile) * self._cost_slope;
369  }
370  }
371  if (AITunnel.IsTunnelTile(new_tile)) {
372  if (AITunnel.GetOtherTunnelEnd(new_tile) != prev_tile) return path.GetCost() + self._cost_tile;
373  return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * self._cost_tile;
374  }
375 
376  /* If the two tiles are more then 1 tile apart, the pathfinder wants a
377  * bridge or tunnel to be build. It isn't an existing bridge / tunnel, as
378  * that case is already handled. */
379  if (AIMap.DistanceManhattan(new_tile, prev_tile) > 1) {
380  /* Check if we should build a bridge or a tunnel. */
381  if (AITunnel.GetOtherTunnelEnd(new_tile) == prev_tile) {
382  return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * (self._cost_tile + self._cost_tunnel_per_tile);
383  } else {
384  return path.GetCost() + AIMap.DistanceManhattan(new_tile, prev_tile) * (self._cost_tile + self._cost_bridge_per_tile) + self._GetBridgeNumSlopes(new_tile, prev_tile) * self._cost_slope;
385  }
386  }
387 
388  /* Check for a turn. We do this by substracting the TileID of the current node from
389  * the TileID of the previous node and comparing that to the difference between the
390  * previous node and the node before that. */
391  local cost = self._cost_tile;
392  if (path.GetParent() != null && (prev_tile - path.GetParent().GetTile()) != (new_tile - prev_tile) &&
393  AIMap.DistanceManhattan(path.GetParent().GetTile(), prev_tile) == 1) {
394  cost += self._cost_turn;
395  }
396 
397  /* Check if the new tile is a coast tile. */
398  if (AITile.IsCoastTile(new_tile)) {
399  cost += self._cost_coast;
400  }
401 
402  /* Check if the last tile was sloped. */
403  if (path.GetParent() != null && !AIBridge.IsBridgeTile(prev_tile) && !AITunnel.IsTunnelTile(prev_tile) &&
404  self._IsSlopedRoad(path.GetParent().GetTile(), prev_tile, new_tile)) {
405  cost += self._cost_slope;
406  }
407 
408  /* Add a cost to "outcost" all paths that aren't using already existing
409  * roads, if that's what we're after */
410  if (!AIRoad.AreRoadTilesConnected(prev_tile, new_tile)) {
411  cost += self._cost_no_existing_road;
412  }
413 
414  /* Add a penalty for road/rail level crossings. */
415  if(AITile.HasTransportType(new_tile, AITile.TRANSPORT_RAIL)) {
416  cost += self._cost_level_crossing;
417  }
418 
419  /* Add a penalty for exisiting drive thru road stations */
420  if(AIRoad.IsDriveThroughRoadStationTile(new_tile)) {
421  cost += self._cost_drivethru_station;
422  }
423 
424  return path.GetCost() + cost;
425 }

427 function _MinchinWeb_RoadPathfinder_::_Estimate(self, cur_tile, cur_direction, goal_tiles)
428 {
429  local min_cost = self._max_cost;
430  /* As estimate we multiply the lowest possible cost for a single tile with
431  * with the minimum number of tiles we need to traverse. */
432  foreach (tile in goal_tiles) {
433  min_cost = min(AIMap.DistanceManhattan(cur_tile, tile) * self._cost_tile * self._distance_penalty, min_cost);
434  }
435  return min_cost;

438 function _MinchinWeb_RoadPathfinder_::_Neighbours(self, path, cur_node)
439 {
440  /* self._max_cost is the maximum path cost, if we go over it, the path isn't valid. */
441  if (path.GetCost() >= self._max_cost) return [];
442  local tiles = [];
443 
444  /* Check if the current tile is part of a bridge or tunnel. */
445  if ((AIBridge.IsBridgeTile(cur_node) || AITunnel.IsTunnelTile(cur_node)) &&
446  AITile.HasTransportType(cur_node, AITile.TRANSPORT_ROAD)) {
447  local other_end = AIBridge.IsBridgeTile(cur_node) ? AIBridge.GetOtherBridgeEnd(cur_node) : AITunnel.GetOtherTunnelEnd(cur_node);
448  local next_tile = cur_node + (cur_node - other_end) / AIMap.DistanceManhattan(cur_node, other_end);
449  if (AIRoad.AreRoadTilesConnected(cur_node, next_tile) || AITile.IsBuildable(next_tile) || AIRoad.IsRoadTile(next_tile)) {
450  tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
451  }
452  /* The other end of the bridge / tunnel is a neighbour. */
453  tiles.push([other_end, self._GetDirection(next_tile, cur_node, true) << 4]);
454  } else if (path.GetParent() != null && AIMap.DistanceManhattan(cur_node, path.GetParent().GetTile()) > 1) {
455  local other_end = path.GetParent().GetTile();
456  local next_tile = cur_node + (cur_node - other_end) / AIMap.DistanceManhattan(cur_node, other_end);
457  if (AIRoad.AreRoadTilesConnected(cur_node, next_tile) || AIRoad.BuildRoad(cur_node, next_tile)) {
458  tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
459  }
460  } else {
461  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
462  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
463  /* Check all tiles adjacent to the current tile. */
464  foreach (offset in offsets) {
465  local next_tile = cur_node + offset;
466  /* We add them to the to the neighbours-list if one of the following applies:
467  * 1) There already is a connections between the current tile and the next tile.
468  * 2) We can build a road to the next tile.
469  * 3) The next tile is the entrance of a tunnel / bridge in the correct direction. */
470  if (AIRoad.AreRoadTilesConnected(cur_node, next_tile)) {
471  tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
472  } else if ((self._cost_only_existing_roads != true) && (AITile.IsBuildable(next_tile) || AIRoad.IsRoadTile(next_tile)) &&
473  (path.GetParent() == null || AIRoad.CanBuildConnectedRoadPartsHere(cur_node, path.GetParent().GetTile(), next_tile)) &&
474  AIRoad.BuildRoad(cur_node, next_tile)) {
475  // WM - add '&& (only_existing_roads != true)' so that non-connected roads are ignored
476  tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
477  } else if ((self._cost_only_existing_roads != true) && self._CheckTunnelBridge(cur_node, next_tile)) {
478  tiles.push([next_tile, self._GetDirection(cur_node, next_tile, false)]);
479  }
480 
481  // Test for water (i.e. rivers or canals or rails to bridge over them
482  local iTile = cur_node + offset;
483  local BridgeLength = 2;
484  while (AITile.HasTransportType(iTile, AITile.TRANSPORT_RAIL) || AITile.IsWaterTile(iTile)) {
485  iTile += offset;
486  BridgeLength++;
487  }
488 
489  // test to see if we could actually build said bridge
490  // TO-DO: Check to see if this test is done elsewhere...
491 
492  if (BridgeLength > 2) {
493  // TO-DO: test for map wraparound... _SuperLib_Tile::IsStraight(tile1, tile2)
494  local BridgeList = AIBridgeList_Length(BridgeLength);
495  if ((BridgeList.Count()) > 0 && (AIBridge.BuildBridge(AIVehicle.VT_ROAD, BridgeList.Begin(), cur_node, iTile))) {
496  local PathCheck = path;
497  local PathParent = path.GetParent();
498  // _MinchinWeb_Log_.Note("Adding Bridge-over tile: " + _MinchinWeb_Array_.ToStringTiles1D([cur_node]) + _MinchinWeb_Array_.ToStringTiles1D([iTile]) + " . " + (self._GetDirection(iTile, cur_node, true) << 4), 7);
499  tiles.push([iTile, self._GetDirection(iTile, cur_node, true) << 4]);
500  }
501  }
502 
503  }
504  if (path.GetParent() != null) {
505  local bridges = self._GetTunnelsBridges(path.GetParent().GetTile(), cur_node, self._GetDirection(path.GetParent().GetTile(), cur_node, true) << 4);
506  foreach (tile in bridges) {
507  tiles.push(tile);
508  }
509  }
510  }
511  return tiles;
512 }

519 function _MinchinWeb_RoadPathfinder_::_GetDirection(from, to, is_bridge)
520 {
521  if (!is_bridge && AITile.GetSlope(to) == AITile.SLOPE_FLAT) return 0xFF;
522  if (from - to == 1) return 1;
523  if (from - to == -1) return 2;
524  if (from - to == AIMap.GetMapSizeX()) return 4;
525  if (from - to == -AIMap.GetMapSizeX()) return 8;
526 
527  // for bridges that don't have a parent tile
528  local direction = from - to;
529  if (direction > 0) {
530  // so direction is positive
531  if (direction < (AIMap.GetMapSizeX() / 2 - 1)) return 1;
532  else return 4;
533  } else {
534  if ((direction * -1) < (AIMap.GetMapSizeX() / 2 - 1)) return 2;
535  else return 8;

546 {
547 // By rights, adding bridge over railroads and water should be added here
548  local slope = AITile.GetSlope(cur_node);
549  if (slope == AITile.SLOPE_FLAT) return [];
550  local tiles = [];
551

553  local bridge_list = AIBridgeList_Length(i + 1);
554  local target = cur_node + i * (cur_node - last_node);

573 function _MinchinWeb_RoadPathfinder_::_IsSlopedRoad(start, middle, end)
574 {
575  local NW = 0; //Set to true if we want to build a road to / from the north-west
576  local NE = 0; //Set to true if we want to build a road to / from the north-east
577  local SW = 0; //Set to true if we want to build a road to / from the south-west
578  local SE = 0; //Set to true if we want to build a road to / from the south-east
579 
580  if (middle - AIMap.GetMapSizeX() == start || middle - AIMap.GetMapSizeX() == end) NW = 1;
581  if (middle - 1 == start || middle - 1 == end) NE = 1;
582  if (middle + AIMap.GetMapSizeX() == start || middle + AIMap.GetMapSizeX() == end) SE = 1;
583  if (middle + 1 == start || middle + 1 == end) SW = 1;
584 
585  /* If there is a turn in the current tile, it can't be sloped. */
586  if ((NW || SE) && (NE || SW)) return false;
587 
588  local slope = AITile.GetSlope(middle);
589  /* A road on a steep slope is always sloped. */
590  if (AITile.IsSteepSlope(slope)) return true;
591 
592  /* If only one corner is raised, the road is sloped. */
593  if (slope == AITile.SLOPE_N || slope == AITile.SLOPE_W) return true;
594  if (slope == AITile.SLOPE_S || slope == AITile.SLOPE_E) return true;
595 
596  if (NW && (slope == AITile.SLOPE_NW || slope == AITile.SLOPE_SE)) return true;
597  if (NE && (slope == AITile.SLOPE_NE || slope == AITile.SLOPE_SW)) return true;
598 
599  return false;
600 }

602 function _MinchinWeb_RoadPathfinder_::_CheckTunnelBridge(current_tile, new_tile)
603 {
604  if (!AIBridge.IsBridgeTile(new_tile) && !AITunnel.IsTunnelTile(new_tile)) return false;
605  local dir = new_tile - current_tile;
606  local other_end = AIBridge.IsBridgeTile(new_tile) ? AIBridge.GetOtherBridgeEnd(new_tile) : AITunnel.GetOtherTunnelEnd(new_tile);
607  local dir2 = other_end - new_tile;
608  if ((dir < 0 && dir2 > 0) || (dir > 0 && dir2 < 0)) return false;
609  dir = abs(dir);
610  dir2 = abs(dir2);
611  if ((dir >= AIMap.GetMapSizeX() && dir2 < AIMap.GetMapSizeX()) ||
612  (dir < AIMap.GetMapSizeX() && dir2 >= AIMap.GetMapSizeX())) return false;
613 
614  return true;
