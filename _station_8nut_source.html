74 function _MinchinWeb_Station_::IsCargoAccepted(StationID, CargoID) {

76  AILog.Warning("MinchinWeb.Station.IsCargoAccepted() was provided with invalid input. Was provided " + StationID + " and " + CargoID + ".");
77  return null;
78  } else {
79  local AllCargos = AICargoList_StationAccepting(StationID);

81  if (AllCargos.HasItem(CargoID)) {
82  return true;
