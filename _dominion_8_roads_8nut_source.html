272 function _MinchinWeb_DLS_::GridPoints(End1, End2) {
273 // ## GridPoints
274 // Returns an array of TileIndexs that are 'grid points' or where roads will
275 // have intersections.
276 // *End1* and *End2* are expected to be TileIndex'es
277 // End1 and End2 will not be included in the return array
278 
279  local x1 = AIMap.GetTileX(End1);
280  local y1 = AIMap.GetTileY(End1);
281  local x2 = AIMap.GetTileX(End2);
282  local y2 = AIMap.GetTileY(End2);
283 
284  if (x1 > x2) {
285  local tempx = x1;
286  x1 = x2;
287  x2 = tempx;
288  }
289  if (y1 > y2) {
290  local tempy = y1;
291  y1 = y2;
292  y2 = tempy;
293  }
294 
295  _MinchinWeb_Log_.Note("Generating grid points from " + x1 + ", " + y1 + " to " + x2 + ", " + y2, 6);
296 
297  // move to first grid x
298  local workingx = x1;
299  while ((workingx - AIMap.GetTileX(this._datum)) % this._gridx != 0) {
300  workingx++;
301  }
302  // move to first grid y
303  local workingy = y1;
304  while ((workingy - AIMap.GetTileY(this._datum)) % this._gridy != 0) {
305  workingy++;
306  }
307 
308  // Cycle through all the grid points and add them to our array
309  // use *do..while* to ensure we get at least one set of grid point per
310  // direction
311  local MyArray = [];
312  local starty = workingy;
313  do {
314  do {
315  local Holding = AIMap.GetTileIndex(workingx, workingy);
316  if ((Holding != End1) && (Holding != End2)) {
317  MyArray.push(Holding);
318  _MinchinWeb_Log_.Note("Add grid point at " + workingx + ", " + workingy, 7)
319  }
320  workingy = workingy + this._gridy;
321  } while (workingy < y2);
322  workingx = workingx + this._gridx;
323  workingy = starty;
324  } while (workingx < x2);
325 
326  _MinchinWeb_Log_.Note("Generated " + MyArray.len() + " grid points.", 6);
327  return MyArray;
328 }

337 // add all grid points between the StartTile and EndTile as intermediate end points
338 // if the pathfinder ends on an intermediate point, make that the new start point and run the pathfinder again

341  cycles = 10000;
342  local StartArray = [];
343  local EndArray = [];

345

347 
348  _MinchinWeb_Log_.Note("+ while loop before: LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 7);
349  do {
350  _MinchinWeb_Log_.Note("++ while loop start: LastTile " + _MinchinWeb_Array_.ToStringTiles1D([LastTile]), 7);

352  EndArray = _MinchinWeb_DLS_.GridPoints(LastTile, this._endtile);
353  EndArray.push(this._endtile);

362  this._running = (Ret == false) ? true : false;
363

367  if ((typeof (TilestoAdd) == "array") && (TilestoAdd.len() > 0)) {
368  TilestoAdd.remove(0); // remove the first tile so we don't get duplicates

379  } while (LastTile != this._endtile)
380

382  this._path = AllTiles;
383  return true;
384  } else {
385  // I don't think we should ever get here...

400 {
401  if (this._running) {
402  AILog.Warning("You can't build a path while there's a running pathfinder.");
403  return false;
404  }
405  if (this._path == null) {
406  AILog.Warning("You have tried to build a 'null' path.");
407  return false;
408  }
409 
410  local TestMode = AIExecMode(); // We're really doing this!
411 
412  AIRoad.SetCurrentRoadType(this._road_type);
413  for (local i=0; i < this._path.len() - 2; i++) {
414  if (AIMap.DistanceManhattan(this._path[i], this._path[i+1]) == 1) {
415  // MD == 1 == road joining the two tiles
416  if (!AIRoad.BuildRoad(this._path[i], this._path[i+1])) {
417  // If we get here, then the road building has failed
418  // Possible that the road already exists
419  // TO-DO:
420  // - fail the road builder if the road cannot be built and
421  // does not already exist
422  // return null;
423  }
424  } else {
425  // Implies that we're building either a tunnel or a bridge
426  if (!AIBridge.IsBridgeTile(this._path[i]) && !AITunnel.IsTunnelTile(this._path[i])) {
427  if (AIRoad.IsRoadTile(this._path[i])) {
428  // Original example demolishes tile if it's already a road
429  // tile to get around expanded roadbits.
430  // I don't like this approach as it could destroy Railway
431  // tracks/tram tracks/station
432  // TO-DO:
433  // - figure out a way to do this while keeping the other
434  // things I've built on the tile
435  // (can I just remove the road?)
436  AITile.DemolishTile(this._path[i]);
437  }
438  if (AITunnel.GetOtherTunnelEnd(this._path[i]) == this._path[i+1]) {
439  // The assumption here is that the land hasn't changed
440  // from when the pathfinder was run and when we try to
441  // build the path. If the tunnel building fails, we
442  // get the 'can't build tunnel' message, but if the
443  // land has changed such that the tunnel end is at a
444  // different spot than is was when the pathfinder ran,
445  // we skip tunnel building and try and build a bridge
446  // instead, which will fail because the slopes are wrong...
447  if (!AITunnel.BuildTunnel(AIVehicle.VT_ROAD, this._path[i])) {
448  // At this point, an error has occurred while building the tunnel.
449  // Fail the pathfiner
450  // return null;
451  AILog.Warning("MinchinWeb.DLS.BuildPath can't build a tunnel from " + AIMap.GetTileX(this._path[i]) + "," + AIMap.GetTileY(this._path[i]) + " to " + AIMap.GetTileX(this._path[i+1]) + "," + AIMap.GetTileY(this._path[i+1]) + "!!" );
452  }
453  } else {
454  // if not a tunnel, we assume we're buildng a bridge
455  local BridgeList = AIBridgeList_Length(AIMap.DistanceManhattan(this._path[i], this._path[i+1] + 1));
456  BridgeList.Valuate(AIBridge.GetMaxSpeed);
457  BridgeList.Sort(AIList.SORT_BY_VALUE, false);
458  if (!AIBridge.BuildBridge(AIVehicle.VT_ROAD, BridgeList.Begin(), this._path[i], this._path[i+1])) {
459  // At this point, an error has occurred while building the bridge.
460  // Fail the pathfiner
461  // return null;
462  AILog.Warning("MinchinWeb.DLS.BuildPath can't build a bridge from " + AIMap.GetTileX(this._path[i]) + "," + AIMap.GetTileY(this._path[i]) + " to " + AIMap.GetTileX(this._path[i+1]) + "," + AIMap.GetTileY(this._path[i+1]) + "!! (or the tunnel end moved...)" );
463  }
464  }
465  }
466  }
467  }

469  // End build sequence
470  return true;
