173 function _MinchinWeb_Marine_::DistanceShip(TileA, TileB) {
174 // Assuming open ocean, ship in OpenTTD will travel 45° angle where possible,
175 // and then finish up the trip by going along a cardinal direction
176  return ((AIMap.DistanceManhattan(TileA, TileB) - AIMap.DistanceMax(TileA, TileB)) * 0.4 + AIMap.DistanceMax(TileA, TileB)).tointeger();
177 }

179 function _MinchinWeb_Marine_::GetPossibleDockTiles(IndustryID) {
180 // Given an industry (by IndustryID), searches for possible tiles to build a
181 // dock and returns the list as an array of TileIndexs
182 
183 // Tiles given should be checked to ensure that the desired cargo is still
184 // accepted
185 
186 // Assumes that the industry location returned is the NE corner of the
187 // industry, and that industries fit within a 4x4 block
188  local Tiles = [];
189  if (AIIndustry.IsValidIndustry(IndustryID) == true) {
190  // Check if the industry already has a dock
191  if (AIIndustry.HasDock(IndustryID) == true) {
192  return [AIIndustry.GetDockLocation(IndustryID)];
193  } else {
194  local ex = AITestMode();
195  local Walker = _MinchinWeb_SW_(); // Spiral Walker
196  Walker.Start(AIIndustry.GetLocation(IndustryID));
197 
198  while (Walker.GetStage() <= ((_MinchinWeb_C_.IndustrySize() + AIStation.GetCoverageRadius(AIStation.STATION_DOCK)) * 4)) {
199  if (AIMarine.BuildDock(Walker.Walk(), AIStation.STATION_NEW) == true) {
200  Tiles.push(Walker.GetTile());
201  }
202  }
203  }
204  _MinchinWeb_Log_.Note("MinchinWeb.Marine.GetPossibleDockTiles() " + _MinchinWeb_Array_.ToStringTiles1D(Tiles, true), 6);
205  return Tiles;
206  } else {
207  AILog.Warning("MinchinWeb.Marine.GetPossibleDockTiles() was supplied with an invalid IndustryID. Was supplied " + IndustryID + ".");
208  return Tiles;
209  }

212 function _MinchinWeb_Marine_::GetDockFrontTiles(Tile) {
213 // Given a tile, returns an array of possible 'front' tiles that a ship could
214 // access the dock from
215 // Can be either the land tile of a dock, or the water tile
216 // Does not test if there is currently a dock at the tile
217 
218 // Tiles under Oil Rigs do NOT return AITile.IsWaterTile(Tile) == true
219 
220 // Might do funny things if the tile given is next to a river (i.e. a flat tile
221 // next to a water tile)
222 
223  local ReturnTiles = [];
224  local offset = AIMap.GetTileIndex(0, 0);;
225  local DockEnd = null;
226  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
227  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
228  local next_tile;
229 
230  if (AIMap.IsValidTile(Tile)) {
231  if (AITile.IsWaterTile(Tile)) {
232  // water tile
233  DockEnd = Tile;
234  } else {
235  // land tile
236  switch (AITile.GetSlope(Tile)) {
237  // see http://vcs.openttd.org/svn/browser/trunk/docs/tileh.png
238  // for slopes
239  case 0:
240  // flat
241  offset = AIMap.GetTileIndex(0, 0);
242  break;
243  case 3:
244  offset = AIMap.GetTileIndex(-1, 0);
245  break;
246  case 6:
247  offset = AIMap.GetTileIndex(0, -1);
248  break;
249  case 9:
250  offset = AIMap.GetTileIndex(0, 1);
251  break;
252  case 12:
253  offset = AIMap.GetTileIndex(1, 0);
254  break;
255  }
256 
257  DockEnd = Tile + offset;
258  }
259 
260  if (DockEnd != null) {
261  /* Check all tiles adjacent to the current tile. */
262  foreach (offset in offsets) {
263  next_tile = DockEnd + offset;
264  if (AITile.IsWaterTile(next_tile)) {
265  ReturnTiles.push(next_tile);
266  }
267  }
268  }
269  } else {
270  AILog.Warning("MinchinWeb.Marine.GetDockFrontTiles() was supplied with an invalid TileIndex. Was supplied " + Tile + ".");
271  }
272 
273  return ReturnTiles;
274 }

276 function _MinchinWeb_Marine_::BuildBuoy(Tile) {
277 // Attempts to build a buoy, but first checks the box within
278 // MinchinWeb.Constants.BuoyOffset() for an existing buoy, and makes sure
279 // there's nothing but water between the two. If no existing buoy is found,
280 // one is built.
281 
282 // Returns the location of the existing or built buoy.
283 
284  local Existing = AITileList();
285  local UseExistingAt = null;
286 
287  local Walker = _MinchinWeb_SW_(); // Spiral Walker
288  Walker.Start(Tile);
289 
290  while (Walker.GetStage() <= (_MinchinWeb_C_.BuoyOffset() * 4)) {
291  if (AIMarine.IsBuoyTile(Walker.Walk())) {
292  Existing.AddItem(Walker.GetTile(), AIMap.DistanceManhattan(Tile, Walker.GetTile()));
293  _MinchinWeb_Log_.Note("BuildBuoy() : Insert Existing at" + _MinchinWeb_Array_.ToStringTiles1D([Walker.GetTile()]), 7);
294  }
295  }
296 
297  if (Existing.Count() > 0) {
298  Existing.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
299  local TestBuoy = Existing.Begin();
300  local KeepTrying = true;
301  local WBC = _MinchinWeb_WBC_();
302  while (KeepTrying == true) {
303  WBC.InitializePath([TestBuoy], [Tile]);
304  WBC.PresetSafety(TestBuoy, Tile);
305  local WBCResults = WBC.FindPath(-1);
306  if (WBCResults != null) {
307  UseExistingAt = TestBuoy;
308  KeepTrying = false;
309  } else {
310  if (Existing.IsEnd()) {
311  KeepTrying = false;
312  UseExistingAt = null;
313  } else {
314  TestBuoy = Existing.Next();
315  }
316  }
317  }
318  }
319 
320  if (UseExistingAt == null) {
321  AIMarine.BuildBuoy(Tile);
322  return Tile;
323  } else {
324  return UseExistingAt;
325  }
326 }

328 function _MinchinWeb_Marine_::BuildDepot(DockTile, Front, NotNextToDock=true) {
329 // Attempts to build a (water) depot, but first checks the box within
330 // MinchinWeb.Constants.WaterDepotOffset() for an existing depot, and makes
331 // sure there's nothing but water between the depot and dock. If no
332 // existing depot is found, one is built.
333 
334 // Returns the location of the existing or built depot.
335 // This will fail if the DockTile given is a dock (or any tile that is not a water tile)
336 
337 // 'NotNextToDock,' when set, will keep the dock from being built next to an
338 // existing dock
339 
340  local StartX = AIMap.GetTileX(DockTile) - _MinchinWeb_C_.WaterDepotOffset();
341  local StartY = AIMap.GetTileY(DockTile) - _MinchinWeb_C_.WaterDepotOffset();
342  local EndX = AIMap.GetTileX(DockTile) + _MinchinWeb_C_.WaterDepotOffset();
343  local EndY = AIMap.GetTileY(DockTile) + _MinchinWeb_C_.WaterDepotOffset();
344 
345  local Existing = AITileList();
346  local UseExistingAt = null;
347 
348  for (local i = StartX; i < EndX; i++) {
349  for (local j = StartY; j < EndY; j++) {
350  if (AIMarine.IsWaterDepotTile(AIMap.GetTileIndex(i,j))) {
351  Existing.AddItem(AIMap.GetTileIndex(i,j), AIMap.DistanceManhattan(DockTile, AIMap.GetTileIndex(i,j)));
352  _MinchinWeb_Log_.Note("BuildDepot() : Insert Existing at" + _MinchinWeb_Array_.ToStringTiles1D([AIMap.GetTileIndex(i,j)]), 6);
353  }
354  }
355  }
356

363  WBC.InitializePath([TestDepot], [DockTile]);
364  WBC.PresetSafety(TestDepot, DockTile);

366  if (WBCResults != null) {

368  KeepTrying = false;

373  } else {
374  TestDepot = Existing.Next();
375  }
376  }
377  }
378  }
379 
380  if (UseExistingAt == null) {
381  if(AIMarine.BuildWaterDepot(DockTile, Front)) {
382  // try and build right at the given spot
383  UseExistingAt = DockTile;
384  } else {
385  // if that doesn't work, build it close by
386  // Generate a list of water tiles, and pick one at random
387  Existing.Clear();
388  for (local i = StartX; i < EndX; i++) {
389  for (local j = StartY; j < EndY; j++) {
390  if (AITile.IsWaterTile(AIMap.GetTileIndex(i,j)) && (_MinchinWeb_Station_.IsNextToDock(AIMap.GetTileIndex(i,j)) == false) ) {
391  Existing.AddItem(AIMap.GetTileIndex(i,j), AIBase.Rand());
392  _MinchinWeb_Log_.Note("BuildDepot() : Insert WaterTile at" + _MinchinWeb_Array_.ToStringTiles1D([AIMap.GetTileIndex(i,j)]), 7);
393  }
394  }
395  }

397  Existing.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
398  local TestDepot = Existing.Begin();

436 function _MinchinWeb_Marine_::RateShips(EngineID, Life, Cargo) {
437 // Designed to Run as a validator
438 // Given the EngineID, it will score them; higher is better
439 // Score = [(Capacity in Cargo)*Reliability*Speed] /
440 // [ (Purchase Price over Life) + (Running Costs)*Life ]
441 //
442 // Life is assumed to be in years
443 // Note: Cargo doesn't work yet. Capacity is measured in the default cargo.
444 
445  local Score = 0;
446  local Age = AIEngine.GetMaxAge(EngineID);
447  local BuyTimes = (Life / Age/366).tointeger() + 1;;
448  // GetMaxAge is given in days
449  local Cost = (BuyTimes * AIEngine.GetPrice(EngineID)) + (Life * AIEngine.GetRunningCost(EngineID)) + 0.001;
450  local Return = (AIEngine.GetCapacity(EngineID) * AIEngine.GetReliability(EngineID) * AIEngine.GetMaxSpeed(EngineID)) + 0.001;
451  if (Return == 0) {
452  Score = 0;
453  } else {
454  Score = (Return * 1000 / Cost).tointeger();
455  }
456 
457  _MinchinWeb_Log_.Note("Rate Ship : " + Score + " : " +AIEngine.GetName(EngineID) + " : " + AIEngine.GetCapacity(EngineID) + " * " + AIEngine.GetReliability(EngineID) + " * " + AIEngine.GetMaxSpeed(EngineID) + " / " + BuyTimes + " * " + AIEngine.GetPrice(EngineID) + " + " + Life + " * " + AIEngine.GetRunningCost(EngineID), 7);
458  return Score;
459 }
