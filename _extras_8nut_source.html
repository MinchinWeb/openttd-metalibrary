241 function _MinchinWeb_Extras_::Slope(TileA, TileB) {
242  local dx = AIMap.GetTileX(TileB) - AIMap.GetTileX(TileA);
243  local dy = AIMap.GetTileY(TileB) - AIMap.GetTileY(TileA);
244 
245  // Zero check
246  if (dx == 0) {
247  return _MinchinWeb_C_.Infinity() * _MinchinWeb_Extras_.Sign(dy);
248  } else if (dy == 0) {
249  return (1.0 / _MinchinWeb_C_.Infinity()) * _MinchinWeb_Extras_.Sign(dx);
250  } else {
251  dx = dx.tofloat();
252  dy = dy.tofloat();
253 
254  return (dy / dx);
255  }
256 }

268  local LowerBound = _MinchinWeb_Extras_.MinFloat(Bound1, Bound2) - _MinchinWeb_C_.FloatOffset();

353 function _MinchinWeb_Extras_::NextCardinalTile(StartTile, TowardsTile) {
354  local Tiles = AITileList();
355  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
356  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
357 
358  foreach (offset in offsets) {
359  Tiles.AddItem(StartTile + offset, AIMap.DistanceSquare(StartTile + offset, TowardsTile));

361 
362  Tiles.Sort(AIList.SORT_BY_VALUE, AIList.SORT_ASCENDING);
363
