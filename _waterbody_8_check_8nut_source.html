184 function _MinchinWeb_WBC_::_Estimate(self, cur_tile, cur_direction, goal_tiles)
185 {
186  local min_cost = self._max_cost;
187  /* As estimate we multiply the lowest possible cost for a single tile with
188  * with the minimum number of tiles we need to traverse. */
189  foreach (tile in goal_tiles) {
190  min_cost = min(AIMap.DistanceManhattan(cur_tile, tile) * self._cost_per_tile * self._distance_penalty, min_cost);
191  }
192  return min_cost;
193 }

195 function _MinchinWeb_WBC_::_Neighbours(self, path, cur_node)
196 {
197  /* self._max_cost is the maximum path cost, if we go over it, the path isn't valid. */
198  if (path.GetCost() >= self._max_cost) return [];
199  local tiles = [];
200 
201  local offsets = [AIMap.GetTileIndex(0, 1), AIMap.GetTileIndex(0, -1),
202  AIMap.GetTileIndex(1, 0), AIMap.GetTileIndex(-1, 0)];
203  /* Check all tiles adjacent to the current tile. */
204  foreach (offset in offsets) {
205  local next_tile = cur_node + offset;
206  if (AIMarine.AreWaterTilesConnected(cur_node, next_tile)) {
207  tiles.push([next_tile, self._GetDirection(cur_node, next_tile)]);
208  }
209  }
210 
214  return tiles;
215 }
